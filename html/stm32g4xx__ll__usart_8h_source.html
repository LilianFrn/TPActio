1930  uint32_t OverSampling)
1931 {
1932  uint32_t usartdiv;
1933  uint32_t brrresult = 0x0U;
1934  uint32_t periphclkpresc = (uint32_t)(PeriphClk / (USART_PRESCALER_TAB[(uint8_t)PrescalerValue]));

1937
